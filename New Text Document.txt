        event_time: Math.floor(Date.now() / 1000),
        event_id: event_id,
        action_source: "website",
        user_data: {
          client_ip_address: req.headers['x-forwarded-for'] || req.socket.remoteAddress,
          client_user_agent: req.headers['user-agent']
        }
      }],
      access_token: "EAARO6HV4aTgBQtO3tWM0CL6kSRnXGUdKyPpMunqclahSfeact7PLrZBwttJZBZBlAPUw9eMxoHnnmXaHoCmWrYOpAx8UIHeH8VKgJE6q7NSmskOU81zlkSwZBj1YRZBAeiF80OLnmpg7btGdypnZCg8NHZBeCSaGLv4ktQzZCZCtEnn7rX0xJAs4zWyEXuT9C2EdUMgZDZD"
    };

    const response = await fetch(
      "https://graph.facebook.com/v19.0/1207170701547324/events",
      {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload)
      }
    );

    const data = await response.json();

    res.status(200).json(data);

  } catch (error) {
    res.status(500).json({ error: error.message });
  }
}
